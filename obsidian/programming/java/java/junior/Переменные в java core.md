> пример инициализации переменных `java` 


```java 
class Test_variable{
	public static void main(String[]args){
		int a ; // инициализация переменной без значения 
		int b =5; // инициализация переменной со значением
		float c =21.6f; // инициализация переменной со значеним (плачающая точка)
		String name ="Andrew"; // инициализация перменной строкового представления
		double myVariable = 0.58386; // инициализация переменной значения double
		boolean is_end_lesson= false;
		System.out.println(b);// вывод значений переменных
		System.out.println("a");// вывод строки в терминал
	}
}
```
