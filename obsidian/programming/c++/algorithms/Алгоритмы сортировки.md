Самый примитивный алгоритм сортировки 
```cpp
int main(){
	const int SIZE = 10;
	int arr[SIZE];
	for(int i=0;i<SIZE;++i){
		for(int j=0;j<SIZE-i-1;j++){
			if (arr[j]>arr[i]){
				int temp = arr[j];
				arr[j]=arr[j+1];
				arr[j+1]=temp;
			}
		}
	}
	for(int i=0;i<SIZE;i++){
		std::cout<<arr[i]<<" ";
	}
}
```
